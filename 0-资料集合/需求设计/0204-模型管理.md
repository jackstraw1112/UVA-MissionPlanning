## 4. 分析流程

### 4.1 模型存储结构设计
- **元数据存储**：使用关系型数据库（如SQLite/PostgreSQL）存储模型元数据，表结构示例：
  ```sql
  CREATE TABLE models (
      id VARCHAR(32) PRIMARY KEY,
      name VARCHAR(255) NOT NULL,
      type VARCHAR(50) NOT NULL,
      version VARCHAR(20) NOT NULL,
      description TEXT,
      creator VARCHAR(100),
      created_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      modified_time TIMESTAMP,
      file_path VARCHAR(500),
      checksum VARCHAR(64),
      tags TEXT,
      status VARCHAR(20) DEFAULT 'active'
  );
  ```
- **文件存储**：模型文件按分类存放在文件系统中，如 `/models/uav/`, `/models/seeker/`，文件名使用 `{id}_{version}.json` 格式。

### 4.2 添加流程
1. 用户上传模型文件，填写元数据表单。
2. 后端计算文件校验和（MD5/SHA256），检查是否已存在相同文件。
3. 生成唯一ID（如UUID），保存文件，插入数据库记录。
4. 返回模型ID。

### 4.3 删除流程
- **物理删除**：直接从数据库删除记录，并删除文件。
- **逻辑删除**：更新 `status` 字段为 `deleted`，文件保留，但列表不显示。

### 4.4 检索流程
- 构建SQL查询，支持模糊匹配（`LIKE %keyword%`）、精确匹配（`type = 'uav'`）、范围查询（`created_time BETWEEN start AND end`）。
- 支持分页：`LIMIT offset, pageSize`。
- 支持排序：`ORDER BY created_time DESC`。

# 反辐射无人机仿真系统——模型管理模块开发实现方案

## 1. 总体架构

模型管理模块采用**元数据+文件存储**分离架构，使用数据库管理模型元数据，文件系统存储模型文件。模块提供完整的增删改查、版本控制、标签分类功能，并与模型构建、校准模块无缝集成。

## 4. 核心功能实现

### 4.1 添加模型

#### 4.1.1 模型文件上传
```cpp
class ModelManager {
public:
    QString addModel(const QString& filePath, const QVariantMap& metadata) {
        // 1. 计算文件校验和
        QFile file(filePath);
        QCryptographicHash hash(QCryptographicHash::Sha256);
        if (file.open(QIODevice::ReadOnly)) {
            hash.addData(&file);
            file.close();
        }
        QString checksum = hash.result().toHex();
        
        // 2. 检查是否已存在相同文件
        if (findModelByChecksum(checksum)) {
            throw std::runtime_error("模型文件已存在");
        }
        
        // 3. 生成唯一ID
        QString modelId = QUuid::createUuid().toString(QUuid::WithoutBraces);
        
        // 4. 确定存储路径
        QString type = metadata["type"].toString();
        QString version = metadata["version"].toString();
        QString fileName = QString("%1_%2.json").arg(modelId).arg(version);
        QString targetDir = QString("./models/%1").arg(type);
        QDir().mkpath(targetDir);
        QString targetPath = targetDir + "/" + fileName;
        
        // 5. 复制文件
        QFile::copy(filePath, targetPath);
        
        // 6. 保存元数据到数据库
        QSqlQuery query;
        query.prepare("INSERT INTO models (id, name, type, version, description, creator, "
                     "file_path, file_size, checksum, tags, metadata) "
                     "VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
        query.addBindValue(modelId);
        query.addBindValue(metadata["name"]);
        query.addBindValue(type);
        query.addBindValue(version);
        query.addBindValue(metadata["description"]);
        query.addBindValue(metadata["creator"]);
        query.addBindValue(targetPath);
        query.addBindValue(QFileInfo(filePath).size());
        query.addBindValue(checksum);
        query.addBindValue(metadata["tags"].toString());
        query.addBindValue(metadata["metadata"].toString());
        query.exec();
        
        return modelId;
    }
};
```

```cpp
bool ModelManager::canDeleteModel(const QString& modelId) {
    QSqlQuery query;
    query.prepare("SELECT COUNT(*) FROM model_references WHERE model_id = ?");
    query.addBindValue(modelId);
    query.exec();
    if (query.next()) {
        return query.value(0).toInt() == 0;
    }
    return true;
}

QStringList ModelManager::getReferences(const QString& modelId) {
    QStringList refs;
    QSqlQuery query;
    query.prepare("SELECT ref_type, ref_id FROM model_references WHERE model_id = ?");
    query.addBindValue(modelId);
    query.exec();
    while (query.next()) {
        refs.append(QString("%1 (%2)").arg(query.value(1).toString())
                                          .arg(query.value(0).toString()));
    }
    return refs;
}
```

#### 4.2.2 执行删除
```cpp
void ModelManager::deleteModel(const QString& modelId, bool permanent = false) {
    if (!canDeleteModel(modelId)) {
        throw std::runtime_error("模型正在被引用，无法删除");
    }
    
    QSqlDatabase::database().transaction();
    try {
        // 获取文件路径
        QSqlQuery query;
        query.prepare("SELECT file_path FROM models WHERE id = ?");
        query.addBindValue(modelId);
        query.exec();
        
        QString filePath;
        if (query.next()) {
            filePath = query.value(0).toString();
        }
        
        if (permanent) {
            // 物理删除
            QFile::remove(filePath);
            query.prepare("DELETE FROM models WHERE id = ?");
        } else {
            // 逻辑删除
            query.prepare("UPDATE models SET status = 'deleted' WHERE id = ?");
        }
        query.addBindValue(modelId);
        query.exec();
        
        // 删除版本历史
        query.prepare("DELETE FROM model_versions WHERE model_id = ?");
        query.addBindValue(modelId);
        query.exec();
        
        QSqlDatabase::database().commit();
    } catch (...) {
        QSqlDatabase::database().rollback();
        throw;
    }
}
```

```cpp
class ModelSearchEngine {
public:
    QVector<ModelInfo> search(const SearchCriteria& criteria) {
        QString sql = "SELECT * FROM models WHERE status = 'active'";
        QVariantList params;
        
        if (!criteria.keyword.isEmpty()) {
            sql += " AND (name LIKE ? OR description LIKE ? OR tags LIKE ?)";
            QString pattern = "%" + criteria.keyword + "%";
            params << pattern << pattern << pattern;
        }
        
        if (!criteria.type.isEmpty()) {
            sql += " AND type = ?";
            params << criteria.type;
        }
        
        if (criteria.startDate.isValid()) {
            sql += " AND created_time >= ?";
            params << criteria.startDate.toString("yyyy-MM-dd HH:mm:ss");
        }
        
        if (criteria.endDate.isValid()) {
            sql += " AND created_time <= ?";
            params << criteria.endDate.toString("yyyy-MM-dd HH:mm:ss");
        }
        
        sql += " ORDER BY created_time DESC LIMIT ? OFFSET ?";
        params << criteria.pageSize << (criteria.page - 1) * criteria.pageSize;
        
        return executeQuery(sql, params);
    }
};
```

### 4.4 版本控制

#### 4.4.1 创建新版本
```cpp
QString ModelManager::createNewVersion(const QString& modelId,
                                        const QString& newVersion,
                                        const QString& comment) {
    // 获取原模型信息
    QSqlQuery query;
    query.prepare("SELECT * FROM models WHERE id = ?");
    query.addBindValue(modelId);
    query.exec();
    
    if (!query.next()) throw std::runtime_error("模型不存在");
    
    // 保存当前版本到历史表
    QSqlQuery insertHist;
    insertHist.prepare("INSERT INTO model_versions (model_id, version, file_path, checksum, comment) "
```

```cpp
class ModelManager {
public:
    QString addModel(const QString& filePath, const QVariantMap& metadata) {
        // 1. 计算文件校验和
        QFile file(filePath);
        QCryptographicHash hash(QCryptographicHash::Sha256);
        if (file.open(QIODevice::ReadOnly)) {
            hash.addData(&file);
            file.close();
        }
        QString checksum = hash.result().toHex();
        
        // 2. 检查是否已存在相同文件
        if (findModelByChecksum(checksum)) {
            throw std::runtime_error("模型文件已存在");
        }
        
        // 3. 生成唯一ID
        QString modelId = QUuid::createUuid().toString(QUuid::WithoutBraces);
        
        // 4. 确定存储路径
        QString type = metadata["type"].toString();
        QString version = metadata["version"].toString();
        QString fileName = QString("%1_%2.json").arg(modelId).arg(version);
        QString targetDir = QString("./models/%1").arg(type);
        QDir().mkpath(targetDir);
        QString targetPath = targetDir + "/" + fileName;
        
        // 5. 复制文件
        QFile::copy(filePath, targetPath);
        
        // 6. 保存元数据到数据库
        QSqlQuery query;
        query.prepare("INSERT INTO models (id, name, type, version, description, creator, "
                     "file_path, file_size, checksum, tags, metadata) "
                     "VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
        query.addBindValue(modelId);
        query.addBindValue(metadata["name"]);
        query.addBindValue(type);
        query.addBindValue(version);
        query.addBindValue(metadata["description"]);
        query.addBindValue(metadata["creator"]);
        query.addBindValue(targetPath);
        query.addBindValue(QFileInfo(filePath).size());
        query.addBindValue(checksum);
        query.addBindValue(metadata["tags"].toString());
        query.addBindValue(metadata["metadata"].toString());
        query.exec();
        
        return modelId;
    }
};
```

```cpp
void ModelManager::deleteModel(const QString& modelId, bool permanent = false) {
    if (!canDeleteModel(modelId)) {
        throw std::runtime_error("模型正在被引用，无法删除");
    }
    
    QSqlDatabase::database().transaction();
    try {
        // 获取文件路径
        QSqlQuery query;
        query.prepare("SELECT file_path FROM models WHERE id = ?");
        query.addBindValue(modelId);
        query.exec();
        
        QString filePath;
        if (query.next()) {
            filePath = query.value(0).toString();
        }
        
        if (permanent) {
            // 物理删除
            QFile::remove(filePath);
            query.prepare("DELETE FROM models WHERE id = ?");
        } else {
            // 逻辑删除
            query.prepare("UPDATE models SET status = 'deleted' WHERE id = ?");
        }
        query.addBindValue(modelId);
        query.exec();
        
        // 删除版本历史
        query.prepare("DELETE FROM model_versions WHERE model_id = ?");
        query.addBindValue(modelId);
        query.exec();
        
        QSqlDatabase::database().commit();
    } catch (...) {
        QSqlDatabase::database().rollback();
        throw;
    }
}
```

```cpp
class ModelSearchEngine {
public:
    QVector<ModelInfo> search(const SearchCriteria& criteria) {
        QString sql = "SELECT * FROM models WHERE status = 'active'";
        QVariantList params;
        
        if (!criteria.keyword.isEmpty()) {
            sql += " AND (name LIKE ? OR description LIKE ? OR tags LIKE ?)";
            QString pattern = "%" + criteria.keyword + "%";
            params << pattern << pattern << pattern;
        }
        
        if (!criteria.type.isEmpty()) {
            sql += " AND type = ?";
            params << criteria.type;
        }
        
        if (criteria.startDate.isValid()) {
            sql += " AND created_time >= ?";
            params << criteria.startDate.toString("yyyy-MM-dd HH:mm:ss");
        }
        
        if (criteria.endDate.isValid()) {
            sql += " AND created_time <= ?";
            params << criteria.endDate.toString("yyyy-MM-dd HH:mm:ss");
        }
        
        sql += " ORDER BY created_time DESC LIMIT ? OFFSET ?";
        params << criteria.pageSize << (criteria.page - 1) * criteria.pageSize;
        
        return executeQuery(sql, params);
    }
};
```
